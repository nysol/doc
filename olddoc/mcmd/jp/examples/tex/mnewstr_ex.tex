\subsubsection*{例1: 基本例}

\begin{Verbatim}[baselinestretch=0.7,frame=single]
$ mnewstr a=attribute,code v=custNo,A0001 l=5 o=rsl1.csv
#END# kgnewstr a=attribute,code l=5 o=rsl1.csv v=custNo,A0001
$ more rsl1.csv

custNo,A0001
custNo,A0001
custNo,A0001
custNo,A0001
\end{Verbatim}
