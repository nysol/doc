\subsubsection*{例1: 基本例}

$str$項目のアルファベット大文字を全て小文字に変換する。


\begin{Verbatim}[baselinestretch=0.7,frame=single]
$ more dat1.csv
id,str
1,ABC
2,aB$12!Cd
3,
4,cBA
$ mcal c='tolower($s{str})' a=rsl i=dat1.csv o=rsl1.csv
#END# kgcal a=rsl c=tolower($s{str}) i=dat1.csv o=rsl1.csv
$ more rsl1.csv
id,str,rsl
1,ABC,abc
2,aB$12!Cd,ab$12!cd
3,,
4,cBA,cba
\end{Verbatim}
