\subsubsection*{例1: 基本例}



\begin{Verbatim}[baselinestretch=0.7,frame=single]
> require 'zdd'
> a=ZDD::itemset("a")
> b=ZDD::itemset("b")
> c=ZDD::itemset("c")
> d=ZDD::itemset("d")
> x=5*a + 3*b + b*c + 2
> x.show
 5 a + b c + 3 b + 2

# アイテムが1つ以下のアイテム集合を選択
> x.permitsym(1).show
 5 a + 3 b + 2

# アイテムが2つ以下のアイテム集合を選択
> x.permitsym(2).show
 5 a + b c + 3 b + 2

# アイテムのないアイテム集合(すなわち空アイテム集合)を選択
> x.permitsym(0).show
 2
\end{Verbatim}
