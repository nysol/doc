\subsubsection*{Example 1: Basic Example}



\begin{Verbatim}[baselinestretch=0.7,frame=single]
> require 'zdd'
> a=ZDD::itemset("a")
> b=ZDD::itemset("b")
> c=ZDD::itemset("c")
> x=5*a+3*b+c
> x.show
 5 a + 3 b + c
> x.maxweight.show
 5

# Obtain the maximum value including constant terms. 
> x=5*a+3*b+c+10
> x.show
 5 a + 3 b + c + 10
> x.maxweight.show
 10

# Select the term with the maximum weight. 
> x=5*a+3*b+5*c+2
> x.show
 5 a + 3 b + 5 c + 2
> x.termsEQ(x.maxweight).show
 5 a + 5 c
\end{Verbatim}
