\subsubsection*{Example 1: Basic Example}



\begin{Verbatim}[baselinestretch=0.7,frame=single]
> require 'zdd'
> a=ZDD::itemset("a")
> b=ZDD::itemset("b")
> c=ZDD::itemset("c")
> x=5*a-3*b+c
> x.show
 5 a - 3 b + c
> x.minweight.show
 - 3

# The maximum constant term is returned. 
> x=5*a-3*b+c-10
> x.show
 5 a - 3 b + c - 10
> x.minweight.show
 - 10

# Select the term with the maximum weight.
> x=5*a-3*b+5*c-3
> x.show
 5 a - 3 b + 5 c - 3
> x.termsEQ(x.minweight).show
 - 3 b - 3
\end{Verbatim}
